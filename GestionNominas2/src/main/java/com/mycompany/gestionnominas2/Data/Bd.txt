
create database gestion_nominas;
use gestion_nominas;


create table empleados (
    dni varchar(15) primary key,
    nombre varchar(100),
    sexo char(1),
    categoria int,
    anyos int
);

create table nominas (
    dni varchar(15) primary key,
    sueldo double,
    foreign key (dni) references empleados(dni)
);
